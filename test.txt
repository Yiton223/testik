local Slider = Shop:CreateSlider({
   Name = "Sell Fruits BELOW KG",
   Range = {0, 10000},
   Increment = 10,
   Suffix = "SellBelowKg",
   CurrentValue = 10,
   Flag = "SliderSellBelowKg", -- A flag is the identifier for the configuration file, make sure every element has a different flag if you're using configuration saving to ensure no overlaps
   Callback = function(Value)
    ShopSettings.SellKG = Value
   end,
})

local Toggle = Shop:CreateToggle({
   Name = "Sell Fruits",
   CurrentValue = false,
   Flag = "Toggle1", -- A flag is the identifier for the configuration file, make sure every element has a different flag if you're using configuration saving to ensure no overlaps
   Callback = function(Value)
    if Value == false then
        task.wait(0.1)
    if Value == true then
        SellFruits()
    end
   end,
})
